[docs]  def pollutant_legend(country, pollutant):
    colormap = Legend.cm.LinearColormap(colors=Legend.country_pollutants[country][pollutant]['color_scale'], 
                                        vmin=0, vmax=Legend.country_pollutants[country][pollutant]['bin_edges'][-1], 
                                        index = Legend.country_pollutants[country][pollutant]['bin_edges'])
    return colormap



[docs]  def pollutant_color_coding(poll, country, pollutant):  
    bin_edges = Legend.country_pollutants[country][pollutant]['bin_edges']  
    idx = np.digitize(poll, bin_edges, right=True)
    return Legend.country_pollutants[country][pollutant]['color_scale'][idx]
